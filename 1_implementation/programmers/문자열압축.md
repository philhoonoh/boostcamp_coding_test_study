# 문자열압축

## Lessons Learned
- 문자열 자르기 

```python
def get_splits(s, split_len):
    return [s[i:i+split_len] for i in range(0, len(s), split_len)]

def cal_len(s_split):
    stack = []
    
    result = ''
    for s in s_split:
        if not stack:
            stack.append(s)
        else:
            if stack[-1] == s:
                stack.append(s)
            elif stack[-1] != s:
                cnt = 0
                expr = stack[-1]
                while stack:
                    stack.pop()
                    cnt += 1
                if cnt != 1:
                    result += str(cnt) + expr
                elif cnt == 1:
                    result += expr
                stack.append(s)

    if stack:
        expr = stack[-1]
        cnt = 0
        while stack:
            stack.pop()
            cnt += 1
        if cnt != 1:
            result += str(cnt) + expr
        elif cnt == 1:
            result += expr

    return result

def solution(s):
    len_s = len(s)
    answer = len_s

    for split_len in range(1, len_s//2+1):
        s_split = get_splits(s, split_len)
        result = cal_len(s_split)
        answer = min(answer, len(result))
        
    return answer
```
